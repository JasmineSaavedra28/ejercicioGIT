function palindromo() { let cadena = prompt("Introduce una palabra para comprobar si es un palindromo o no"); let palabra = cadena.toLowerCase().replace(/\s/g,""); let alreves = palabra.split("").reverse().join(""); if (palabra == alreves){ console.log("La cadena es un palíndromo"); }else{ console.log("La cadena no es un palíndromo"); } } palindromo();
